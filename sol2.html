    #include <iostream>
    #include <vector>
    #include <algorithm>
    #include <unordered_map>
    #include <limits.h>
    
    using namespace std;
    int getMin(unordered_map<int, int> &myMap)
    {
        int minValue = INT_MAX;
        for (auto it : myMap)
        {
            if (it.second != 0)
            {
                minValue = min(minValue, it.first);
            }
        }
        return minValue;
    }
    
    bool isNStraightHand(vector<int> &hand, int groupsize)
    {
        int len = hand.size();
        if (len % groupsize != 0)
            return false;
    
        sort(hand.begin(), hand.end());
    
        unordered_map<int, int> count;
        for (int n : hand)
        {
            count[n]++;
        }
    
        while (!count.empty())
        {
            int start = getMin(count);
            if (start == INT_MAX)
                break;
    
            for (int i = 0; i < groupsize; i++)
            {
                if (count.find(start + i) == count.end() || count[start + i] == 0)
                {
                    return false;
                }
                count[start + i]--;
                if (count[start + i] == 0)
                {
                    count.erase(start + i);
                }
            }
        }
    
        return true;
    }
